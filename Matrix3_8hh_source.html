87  // if quaternion is non-unit, creates rotation corresponding to the normalized(unit) quaternion

376 // Returns the rotation angle and rotation axis(Geant4).
377  //rotation defined by an angle and a vector
378  void AngleAxis(Angle& angle, Vector3<Value_>& axis) const {
379  auto cosa = 0.5 * (x_.X() + y_.Y() + z_.Z() - 1);
380  auto cosa1 = 1 - cosa;
381  if (cosa1 <= Value_(0)) {
382  angle = 0_rad;
383  axis = Vector3<Value_>(0, 0, 1);
384  } else {
385  auto x = 0.;
386  auto y = 0.;
387  auto z = 0.;
388  if (x_.X() > cosa) x = std::sqrt((x_.X() - cosa) / cosa1);
389  if (y_.Y() > cosa) y = std::sqrt((y_.Y() - cosa) / cosa1);
390  if (z_.Z() > cosa) z = std::sqrt((z_.Z() - cosa) / cosa1);
391  if (z_.Y() < y_.Z()) x = -x;
392  if (x_.Z() < z_.X()) y = -y;
393  if (y_.X() < x_.Y()) z = -z;
394  angle = boost::units::acos(cosa);
395  axis = Vector3<Value_>(x, y, z);
396  }
397  }

417  Angle XPhi() const {
418  auto s2 = ValueSquare(1) - z_.Z() * z_.Z();
419  if (s2 < ValueSquare(0)) {
420  std::cout << "Phi() |z_.Z()| > 1 " << std::endl;
421  s2 = 0;
422  }
423  auto sinTheta = std::sqrt(s2);
424  if (sinTheta != Value_(0)) {
425  auto cscTheta = 1 / sinTheta;
426  auto cosAbsPhi = z_.Y() * cscTheta;
427  if (std::abs(cosAbsPhi) > 1) { // NaN-proofing
428  std::cout << "Phi() finds | cos phi | > 1" << std::endl;
429  cosAbsPhi = 1;
430  }
431  auto absPhi = acos(cosAbsPhi);
432  if (z_.X() > 0) return absPhi;
433  if (z_.X() < 0) return -absPhi;
434  if (z_.Y() > 0) return 0_rad;
435  return Pi();
436  } else { // sinTheta == Value(0) so |Fzz| = 1
437  auto absPhi = .5 * boost::units::acos(x_.X());
438  if (x_.Y() > 0) return -absPhi;
439  if (x_.Y() < 0) return absPhi;

508  auto SubMatrix(Dim3 delete_1, Dim3 delete_2) const {
509  auto dim2_1 = EnumIterator<Dim2>{Dim2::x};

511  auto matrix = Matrix2<Value_>{};

574 // Take two input vectors(in xAxis, and zAxis) and turn them into an
575 // orthogonal basis. This is an internal helper function used to implement
576 // the Set?Axis functions, but is exposed because the functionality is
577 // often useful.
578  Matrix3 MakeBasis(Vector3<Value_> const& xAxis, Vector3<Value_> const& zAxis) const {
579  auto matrix = Matrix3{};
580  // Make the Z axis into a unit variable.
581  auto zmag = zAxis.Mag();
582  if (zmag < 1E-6) std::cout << "MakeBasis(X,Y,Z) non-zero Z Axis is required" << std::endl;
583  matrix.Z() = zAxis / zmag;
584  auto xmag = xAxis.Mag();
585  if (xmag < 1E-6 * zmag) {
586  matrix.X() = matrix.Z().Orthogonal();
587  xmag = 1;
588  } else matrix.X() = xAxis;
589  // Find the Y axis
590  matrix.Y() = matrix.Z().Cross(matrix.X()) * (1 / xmag);
591  auto ymag = matrix.Y().Mag();
592  matrix.Y() = ymag < 1E-6 * zmag ? matrix.Z().Orthogonal() : matrix.Y() / ymag;
593  matrix.X() = matrix.Y().Cross(matrix.Z());
594  return matrix;
595  }
