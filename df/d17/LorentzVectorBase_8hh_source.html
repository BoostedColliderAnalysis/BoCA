299 // Lorentz boost.
300  void Boost(Vector3<double> const& boost) {
301  //Boost this Lorentz vector
302  auto mag_2 = boost.Mag2();
303  auto gamma = 1. / std::sqrt(1. - mag_2);
304  auto bp = boost * Vector();
305  auto gamma2 = mag_2 > 0. ? (gamma - 1.) / mag_2 : 0.;
306  Vector() = Vector() + gamma2 * bp * boost + gamma * boost * T();
307  T() = gamma * (T() + bp);
308  }

310  LorentzVectorBase<Value_> Boosted(Vector3<double> const& boost) const {
311  //Boost this Lorentz vector
312  auto mag_2 = boost.Mag2();
313  auto gamma = 1. / std::sqrt(1. - mag_2);
314  auto gamma2 = mag_2 > 0. ? (gamma - 1.) / mag_2 : 0.;
315  auto bp = boost * Vector();
316  auto lorentz_vector = LorentzVectorBase<Value_>{};
317  lorentz_vector.Vector() = Vector() + gamma2 * bp * boost + gamma * boost * T();
318  lorentz_vector.SetT(gamma * (T() + bp));
319  return lorentz_vector;
320  }

329  boca::Angle Rapidity(Vector3<double> const& ref) const {
330  auto r = ref.Mag2();
331  if (r == 0) {
332  std::cout << "A zero vector used as reference to LorentzVector rapidity" << std::endl;
333  return 0;
334  }
335  auto vdotu = Vector().Dot(ref) / std::sqrt(r);
336  if (vdotu == Value_(0)) return 0_rad;
337  if (T() <= Value_(0)) std::cout << "Tried to take rapidity of negative-energy Lorentz vector" << std::endl;
338  auto pt = sqrt(units::max(sqr(T() * std::numeric_limits<double>::epsilon()), Perp2(ref) + Mag2()));
339  auto rap = units::log(((T() + abs(Z())) / pt).value());
340  return Z() > Value_(0) ? rap : -rap;
341  }
