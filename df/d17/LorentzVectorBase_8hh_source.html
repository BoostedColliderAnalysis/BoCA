333 // Lorentz boost.
334  void Boost(Value bx, Value by, Value bz) {
335  //Boost this Lorentz vector
336  ValueSqr b2 = bx * bx + by * by + bz * bz;
337  Value gamma = 1.0 / std::sqrt(1.0 - b2);
338  ValueSqr bp = bx * X() + by * Y() + bz * Z();
339  Value gamma2 = b2 > 0 ? (gamma - 1.0) / b2 : 0.0;
340  SetX(X() + gamma2 * bp * bx + gamma * bx * T());
341  SetY(Y() + gamma2 * bp * by + gamma * by * T());
342  SetZ(Z() + gamma2 * bp * bz + gamma * bz * T());
343  SetT(gamma * (T() + bp));
344  }
345  void Boost(Vector3<Value> const& b) {
346  Boost(b.X(), b.Y(), b.Z());
347  }
