301 // The angle w.r.t. another 3-vector.
302  boca::Angle Angle(Vector3 const& vector) const {
303  // return the angle w.r.t. another 3-vector
304  auto ptot2 = Mag2() * vector.Mag2();
305  if (ptot2 <= Value4(0)) return 0_rad;
306  auto arg = Dot(vector) / sqrt(ptot2);
307  if (arg > 1) arg = 1;
308  if (arg < -1) arg = -1;
309 
310  auto res = acos(arg);
311 
312  // TVector3 finds the angle by acos(one.two / sqrt(one.one*two.two))
313  // This has a relative error of 0.5*epsilon / angle**2 (where epsilon is machine epsilon)
314  // due to catastrophic cancellation
315  // This form is better (from my personal experiments)
316  auto test = atan2(Cross(vector).Mag(), Dot(vector));
317 
318  if(res != test) Debug("first", res, "second", test);
319  return boca::Angle(res);
320  }

322  // Find the tan(theta)**2 between two PseudoJets by calculating (p3>_1 x p3>_2)**2 / (p3>_1 . p3>_2)**2
323  double Tan2(Vector3 const& vector){
324  // This is more accurate than an alternate form (one.one*two.two-(one.two)**2)/(one.two)**2
325  // because the cancellation in the numerator (cross-product) is handled component-by-component, rather than all at once.
326  auto numerator = sqr(X() * vector.Y() - Y() * vector.X());
327  numerator += sqr(X() * vector.Z() - Z() * vector.X());
328  numerator += sqr(Y() * vector.Z() - Z() * vector.Y());
329  return numerator / sqr(Dot(vector));
330  }

380 // Rotates reference frame from Uz to newUz(unit vector)(Geant4).
381  void RotateUz(Vector3 const& vector) {
382  // NewUzVector must be normalized !
383  auto x = vector.x_;
384  auto y = vector.y_;
385  auto z = vector.z_;
386  auto square = sqr(x) + sqr(y);
387  if (square) {
388  auto sqrt = std::sqrt(square);
389  auto px = x_;
390  auto py = y_;
391  auto pz = z_;
392  x_ = (x * z * px - y * py + x * sqrt * pz) / sqrt;
393  y_ = (y * z * px + x * py + y * sqrt * pz) / sqrt;
394  z_ = (z * z * px - px + z * sqrt * pz) / sqrt;
395  } else if (z < Value(0)) {
396  x_ = -x_; // phi=0 teta=pi
397  z_ = -z_;
398  } else {};
399  }
