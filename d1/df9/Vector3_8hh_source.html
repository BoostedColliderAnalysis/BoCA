188  template <typename Value_2>
189  ValueSquared Perp2(Vector3<Value_2> const& vector) const {
190  auto other_mag2 = vector.Mag2();
191  auto mixing = Dot(vector);
192  ValueSquared this_mag_2 = Mag2();
193  if (other_mag2 > boca::ValueProduct<Value_2, Value_2>(0)) this_mag_2 -= sqr(mixing) / other_mag2;
194  if (this_mag_2 < ValueSquared(0)) this_mag_2 = ValueSquared(0);

239 // The angle w.r.t. another 3-vector.
240  boca::Angle Angle(Vector3 const& vector) const {
241  // return the angle w.r.t. another 3-vector
242  Value ptot2 = Mag2() * vector.Mag2();
243  if (ptot2 <= 0) return 0. * rad;
244  else {
245  Value arg = Dot(vector) / std::sqrt(ptot2);
246  if (arg > 1) arg = 1;
247  if (arg < -1) arg = -1;
248  return std::acos(arg) * rad;

300 // Rotates reference frame from Uz to newUz(unit vector)(Geant4).
301  void RotateUz(Vector3 const& vector) {
302  // NewUzVector must be normalized !
303  Value x = vector.x_;
304  Value y = vector.y_;
305  Value z = vector.z_;
306  ValueSquared square = sqr(x) + sqr(y);
307  if (square) {
308  square = std::sqrt(square);
309  Value px = x_;
310  Value py = y_;
311  Value pz = z_;
312  x_ = (x * z * px - y * py + x * square * pz) / square;
313  y_ = (y * z * px + x * py + y * square * pz) / square;
314  z_ = (z * z * px - px + z * square * pz) / square;
315  } else if (z < 0.) {
316  x_ = -x_; // phi=0 teta=pi
317  z_ = -z_;
318  } else {};
319  }
