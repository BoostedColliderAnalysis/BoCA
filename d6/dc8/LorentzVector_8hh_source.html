102  // Transverse energy w.r.t. given axis squared.
103  template<typename Value, typename = OnlyIfNotQuantity<Value>>
104  EnergySquare Et2(Vector3<Value> const& vector) const {
105  auto pt2 = vector_3_.Perp2(vector);
106  auto pv = vector_3_.Dot(vector.Unit());
107  return pt2 == 0. * GeV2 ? 0. * GeV2 : sqr(E()) * pt2 / (pt2 + sqr(pv));
108  }
109 
110  // Transverse energy w.r.t. given axis.
111  template<typename Value, typename = OnlyIfNotQuantity<Value>>
112  boca::Energy Et(Vector3<Value> const& vector) const {
113  auto etet = Et2(vector);
114  return E() < 0_GeV ? -sqrt(etet) : sqrt(etet);
115  }
