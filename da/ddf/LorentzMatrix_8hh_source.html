151  Matrix3<Value_> SubMatrix(LorentzDim delete_1, LorentzDim delete_2) const {
152 // std::cout << "delete " << Name(delete_1) << " " << Name(delete_2) << std::endl;
153  auto dim3_1 = EnumIterator<Dim3>{Dim3::x};
154  auto dim3_2 = EnumIterator<Dim3>{Dim3::x};
155  auto matrix = Matrix3<Value_>{};
156  for (auto dim4_1 : LorentzDimensions()) {
157  if (dim4_1 == delete_1) continue;
158  for (auto dim4_2 : LorentzDimensions()) {
159  if (dim4_2 == delete_2) continue;
160 // std::cout << "insert " << Name(dim4_1) << " " << Name(dim4_2) << " into "<< Name(*dim3_1) << " " << Name(*dim3_2) << std::endl;
161  matrix(*dim3_1, *dim3_2) = (*this)(dim4_1, dim4_2);
162  ++dim3_2;
163  }
164  ++dim3_1;
165  dim3_2.Set(Dim3::x);
166  }
167  return matrix;
168  }

350  // Set elements according to a boost vector.
351  //boost this Lorentz vector
352  void SetBoost(Vector3<double> const& boost) {
353  auto bp2 = boost.Mag2();
354  auto gamma = 1 / std::sqrt(1 - bp2);
355  auto bgamma = sqr(gamma) / (1 + gamma);
356  auto matrix = Matrix3<double>(bgamma, Matrix::uniform) + Multiply(boost, boost) + Matrix3<double>(1);
357  auto vector = boost + Vector3<double>(gamma, gamma, gamma);
358  *this = LorentzMatrix<double>(matrix, vector, gamma);
359  }
