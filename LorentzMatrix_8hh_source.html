155  Matrix3<Value_> SubMatrix(LorentzDim delete_1, LorentzDim delete_2) const {
156 // std::cout << "delete " << Name(delete_1) << " " << Name(delete_2) << std::endl;
157  auto dim3_1 = EnumIterator<Dim3>{Dim3::x};
158  auto dim3_2 = EnumIterator<Dim3>{Dim3::x};
159  auto matrix = Matrix3<Value_>{};
160  for (auto dim4_1 : LorentzDimensions()) {
161  if (dim4_1 == delete_1) continue;
162  for (auto dim4_2 : LorentzDimensions()) {
163  if (dim4_2 == delete_2) continue;
164 // std::cout << "insert " << Name(dim4_1) << " " << Name(dim4_2) << " into "<< Name(*dim3_1) << " " << Name(*dim3_2) << std::endl;
165  matrix(*dim3_1, *dim3_2) = (*this)(dim4_1, dim4_2);
166  ++dim3_2;
167  }
168  ++dim3_1;
169  dim3_2.Set(Dim3::x);
170  }
171  return matrix;
172  }

354  // Set elements according to a boost vector.
355  //boost this Lorentz vector
356  void SetBoost(Vector3<double> const& boost) {
357  auto bp2 = boost.Mag2();
358  auto gamma = 1 / std::sqrt(1 - bp2);
359  auto bgamma = sqr(gamma) / (1 + gamma);
360  auto matrix = Matrix3<double>(bgamma, Matrix::uniform) + Multiply(boost, boost) + Matrix3<double>(1);
361  auto vector = boost + Vector3<double>(gamma, gamma, gamma);
362  *this = LorentzMatrix<double>(matrix, vector, gamma);
363  }
