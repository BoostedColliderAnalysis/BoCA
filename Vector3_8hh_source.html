322  boca::Angle Angle(Vector3 const &vector) const
323  {
324  // return the angle w.r.t. another 3-vector
325  auto pmag2 = Mag2() * vector.Mag2();
326  if (pmag2 <= Value4(0)) return 0_rad;
327  auto arg = Dot(vector) / sqrt(pmag2);
328  if (arg > 1) arg = 1;
329  if (arg < -1) arg = -1;
330 
331  auto res = acos(arg);
332 
333  // TVector3 finds the angle by acos(one.two / sqrt(one.one*two.two))
334  // This has a relative error of 0.5*epsilon / angle**2 (where epsilon is machine epsilon)
335  // due to catastrophic cancellation
336  // This form is better (from my personal experiments)
337  auto test = atan2(Cross(vector).Mag(), Dot(vector));
338 
339  if (res != test) Debug("first", res, "second", test);
340  return boca::Angle(res);
341  }
