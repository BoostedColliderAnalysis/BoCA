290  auto other_mag2 = vector.Mag2();

292  auto this_mag_2 = Mag2();

349  Vector3 Orthogonal() const {
350  auto x = x_ < Value_(0) ? -x_ : x_;

357  using Value4 = typename boost::units::multiply_typeof_helper<ValueSquare, ValueSquare>::type;
358 // The angle w.r.t. another 3-vector.
359  boca::Angle Angle(Vector3 const& vector) const {
360  // return the angle w.r.t. another 3-vector
361  auto ptot2 = Mag2() * vector.Mag2();
362  if (ptot2 <= Value4(0)) return 0_rad;
363  auto arg = Dot(vector) / sqrt(ptot2);
364  if (arg > 1) arg = 1;
365  if (arg < -1) arg = -1;
366 
367  auto res = acos(arg);
368 
369  // TVector3 finds the angle by acos(one.two / sqrt(one.one*two.two))
370  // This has a relative error of 0.5*epsilon / angle**2 (where epsilon is machine epsilon)
371  // due to catastrophic cancellation
372  // This form is better (from my personal experiments)
373  auto test = atan2(Cross(vector).Mag(), Dot(vector));
374 
375  if(res != test) Debug("first", res, "second", test);
376  return boca::Angle(res);
377  }

379  // Find the tan(theta)**2 between two PseudoJets by calculating (p3>_1 x p3>_2)**2 / (p3>_1 . p3>_2)**2
380  double Tan2(Vector3 const& vector) {
381  // This is more accurate than an alternate form (one.one*two.two-(one.two)**2)/(one.two)**2
382  // because the cancellation in the numerator (cross-product) is handled component-by-component, rather than all at once.
383  auto numerator = sqr(X() * vector.Y() - Y() * vector.X());
384  numerator += sqr(X() * vector.Z() - Z() * vector.X());
385  numerator += sqr(Y() * vector.Z() - Z() * vector.Y());
386  return numerator / sqr(Dot(vector));
387  }

437 // Rotates reference frame from Uz to newUz(unit vector)(Geant4).
438  void RotateUz(Vector3 const& vector) {
439  // NewUzVector must be normalized !
440  auto x = vector.x_;
441  auto y = vector.y_;
442  auto z = vector.z_;
443  auto square = sqr(x) + sqr(y);
444  if (square) {
445  auto sqrt = std::sqrt(square);
446  auto px = x_;
447  auto py = y_;
448  auto pz = z_;
449  x_ = (x * z * px - y * py + x * sqrt * pz) / sqrt;
450  y_ = (y * z * px + x * py + y * sqrt * pz) / sqrt;
451  z_ = (z * z * px - px + z * sqrt * pz) / sqrt;
452  } else if (z < Value_(0)) {
453  x_ = -x_; // phi=0 teta=pi
454  z_ = -z_;
455  } else {};
456  }
